@ViewChildren — QueryList Methods

this.userItems.length // total count — like array.length
this.userItems.first // first item
this.userItems.last // last item
this.userItems.forEach(...) // loop through all
this.userItems.toArray() // convert to regular array
this.userItems.changes // Observable — fires when list changes

static: true vs static: false

// static: false (DEFAULT) — available in ngAfterViewInit
@ViewChild('myInput') myInput!: ElementRef;
// static: true — available in ngOnInit
// use ONLY when element is always in DOM (not inside *ngIf)
@ViewChild('myInput', static: true ) myInput!: ElementRef;

Quick Summary

@ViewChild @ViewChildren
Gets ONE element ALL matching elements
Returns ElementRef or Component QueryList
Access in ngAfterViewInit ngAfterViewInit
React equivalent useRef() array of refs
Use for single DOM/component multiple same elements
